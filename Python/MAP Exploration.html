df_MAP = pd.read_csv(MAP, delimiter = ',')
df_ib = pd.read_csv(ib_results, delimiter = ',')

df_map_ib = df_MAP.merge(df_ib, how='left', left_on='student_id', right_on='StudentID')

def ib_case(row):
    if row['ib_diploma_result'] == 1:
        return 'Diploma Awarded'
    elif row['ib_diploma_result'] == 0:
        return 'Diploma Not Awarded'
    elif row['ib_diploma_result'] == 2:
        return 'Not Enrolled in DP'
    else:
        return NaN
    
df_map_ib['ib_result'] = df_map_ib.apply(ib_case, axis=1)
